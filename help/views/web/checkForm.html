表单验证
表达验证用于区域内表单的自动化验证，使用 module.checkForm(form) form 验证区域父级jquery对象，必填项 请在input元素添加required属性，对应不同规则的验证，采用配置的方式，方便后期扩展，使用方法是 配置data-type属性，现有验证属性如下： 
tel: /^1[3|4|5|7|8][0-9]\d{8}$/,		/*! 匹配手机号 */
email: /^(\w)+(\.\w)*@(\w)+((\.\w+)+)$/,	/* ! 匹配email */
chars: /^[\u4E00-\u9FA5a-zA-Z0-9]{1,}$/,	/* ! 禁止输入特殊字符 */
card: /(^\d{15}$)|(^\d{18}$)|(^\d{17}(\d|X|x)$)/, /* ! 匹配身份证号码 */
int: /^[0-9]*$/,					/* ! 匹配正整数 */
pwd: /^[a-zA-Z]\w{5,17}$/,			/* ! 以字母开头，长度在6-18之间，只能包含字符、数字和下划线 密码 */
num: /^[0-9]+(.[0-9]{1,})?$/,	/* ! 匹配整数和小数 */
